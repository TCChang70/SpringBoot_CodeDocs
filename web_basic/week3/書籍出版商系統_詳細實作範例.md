```java
package model;

import java.io.Serializable;
import javax.persistence.*;
import java.util.List;
import java.util.ArrayList;
import com.fasterxml.jackson.annotation.JsonManagedReference;

/**
 * 出版商實體類別 - 一對多關聯的"一"端
 * 一個出版商可以出版多本書籍
 */
@Entity
@Table(name="publishers")
@NamedQueries({
    @NamedQuery(name="Publisher.findAll", query="SELECT p FROM Publisher p"),
    @NamedQuery(name="Publisher.findByName", query="SELECT p FROM Publisher p WHERE p.publisherName LIKE :name"),
    @NamedQuery(name="Publisher.findByCountry", query="SELECT p FROM Publisher p WHERE p.country = :country")
})
public class Publisher implements Serializable {
    private static final long serialVersionUID = 1L;

    @Id
    @Column(name="publisher_id", length=10)
    private String publisherId;

    @Column(name="publisher_name", nullable=false, length=100)
    private String publisherName;

    @Column(length=200)
    private String address;

    @Column(length=50)
    private String city;

    @Column(length=50)
    private String country;

    @Column(length=20)
    private String phone;

    @Column(length=100)
    private String email;

    @Column(length=100)
    private String website;

    // 一對多雙向關聯配置詳解
    @OneToMany(
        mappedBy="publisher",           // 指定關聯的擁有方是Book實體中的publisher屬性
        cascade=CascadeType.ALL,        // 所有操作都會級聯到子實體
        fetch=FetchType.LAZY,           // 延遲載入，提升效能
        orphanRemoval=true              // 當書籍從集合中移除時，自動刪除該書籍
    )
    @JsonManagedReference              // 解決JSON循環參照問題的主控端
    private List<Book> books = new ArrayList<>();

    // 建構子
    public Publisher() {
    }

    public Publisher(String publisherId, String publisherName) {
        this.publisherId = publisherId;
        this.publisherName = publisherName;
    }

    public Publisher(String publisherId, String publisherName, String address, 
                    String city, String country, String phone, String email, String website) {
        this.publisherId = publisherId;
        this.publisherName = publisherName;
        this.address = address;
        this.city = city;
        this.country = country;
        this.phone = phone;
        this.email = email;
        this.website = website;
    }

    // === Getter 和 Setter 方法 ===
    public String getPublisherId() {
        return this.publisherId;
    }

    public void setPublisherId(String publisherId) {
        this.publisherId = publisherId;
    }

    public String getPublisherName() {
        return this.publisherName;
    }

    public void setPublisherName(String publisherName) {
        this.publisherName = publisherName;
    }

    public String getAddress() {
        return this.address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public String getCity() {
        return this.city;
    }

    public void setCity(String city) {
        this.city = city;
    }

    public String getCountry() {
        return this.country;
    }

    public void setCountry(String country) {
        this.country = country;
    }

    public String getPhone() {
        return this.phone;
    }

    public void setPhone(String phone) {
        this.phone = phone;
    }

    public String getEmail() {
        return this.email;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    public String getWebsite() {
        return this.website;
    }

    public void setWebsite(String website) {
        this.website = website;
    }

    public List<Book> getBooks() {
        return this.books;
    }

    public void setBooks(List<Book> books) {
        this.books = books;
    }

    // === 關聯維護的便利方法 ===
    
    /**
     * 新增書籍 - 維護雙向關聯
     * @param book 要新增的書籍
     * @return 新增的書籍
     */
    public Book addBook(Book book) {
        if (this.books == null) {
            this.books = new ArrayList<>();
        }
        this.books.add(book);
        book.setPublisher(this);  // 設定反向關聯
        return book;
    }

    /**
     * 移除書籍 - 維護雙向關聯
     * @param book 要移除的書籍
     * @return 移除的書籍
     */
    public Book removeBook(Book book) {
        if (this.books != null) {
            this.books.remove(book);
        }
        book.setPublisher(null);  // 清除反向關聯
        return book;
    }

    /**
     * 取得書籍數量
     * @return 該出版商的書籍總數
     */
    public int getBookCount() {
        return this.books != null ? this.books.size() : 0;
    }

    /**
     * 檢查是否有指定的書籍
     * @param book 要檢查的書籍
     * @return 是否包含該書籍
     */
    public boolean hasBook(Book book) {
        return this.books != null && this.books.contains(book);
    }

    // === 業務邏輯方法 ===
    
    /**
     * 計算所有書籍的總價值
     * @return 所有書籍價格的總和
     */
    public java.math.BigDecimal getTotalBookValue() {
        if (this.books == null || this.books.isEmpty()) {
            return java.math.BigDecimal.ZERO;
        }
        
        return this.books.stream()
            .map(Book::getPrice)
            .filter(price -> price != null)
            .reduce(java.math.BigDecimal.ZERO, java.math.BigDecimal::add);
    }

    /**
     * 取得指定分類的書籍
     * @param category 書籍分類
     * @return 該分類的書籍列表
     */
    public List<Book> getBooksByCategory(String category) {
        if (this.books == null) {
            return new ArrayList<>();
        }
        
        return this.books.stream()
            .filter(book -> category.equals(book.getCategory()))
            .collect(java.util.stream.Collectors.toList());
    }

    @Override
    public String toString() {
        return "Publisher [publisherId=" + publisherId + ", publisherName=" + publisherName 
                + ", city=" + city + ", country=" + country + ", bookCount=" + getBookCount() + "]";
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) return true;
        if (obj == null || getClass() != obj.getClass()) return false;
        Publisher publisher = (Publisher) obj;
        return publisherId != null ? publisherId.equals(publisher.publisherId) : publisher.publisherId == null;
    }

    @Override
    public int hashCode() {
        return publisherId != null ? publisherId.hashCode() : 0;
    }
}
```

### Book.java - 書籍實體（完整版）

```java
package model;

import java.io.Serializable;
import javax.persistence.*;
import java.math.BigDecimal;
import java.time.LocalDate;
import com.fasterxml.jackson.annotation.JsonBackReference;
import com.fasterxml.jackson.annotation.JsonFormat;

/**
 * 書籍實體類別 - 一對多關聯的"多"端
 * 多本書籍屬於一個出版商
 */
@Entity
@Table(name="books")
@NamedQueries({
    @NamedQuery(name="Book.findAll", query="SELECT b FROM Book b ORDER BY b.publishDate DESC"),
    @NamedQuery(name="Book.findByTitle", query="SELECT b FROM Book b WHERE b.title LIKE :title"),
    @NamedQuery(name="Book.findByAuthor", query="SELECT b FROM Book b WHERE b.author LIKE :author"),
    @NamedQuery(name="Book.findByCategory", query="SELECT b FROM Book b WHERE b.category = :category"),
    @NamedQuery(name="Book.findByISBN", query="SELECT b FROM Book b WHERE b.isbn = :isbn"),
    @NamedQuery(name="Book.findByPriceRange", 
                query="SELECT b FROM Book b WHERE b.price BETWEEN :minPrice AND :maxPrice")
})
public class Book implements Serializable {
    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy=GenerationType.IDENTITY)
    @Column(name="book_id")
    private int bookId;

    @Column(nullable=false, length=200)
    private String title;

    @Column(nullable=false, length=100)
    private String author;

    @Column(unique=true, length=20)
    private String isbn;

    @Column(precision=10, scale=2)
    private BigDecimal price;

    @Column(name="publish_date")
    @JsonFormat(pattern="yyyy-MM-dd")  // JSON 日期格式化
    private LocalDate publishDate;

    @Column(length=50)
    private String category;

    @Column(name="page_count")
    private Integer pageCount;

    @Column(length=1000)
    private String description;

    @Column(name="stock_quantity")
    private Integer stockQuantity = 0;

    // 多對一雙向關聯配置詳解
    @ManyToOne(
        fetch=FetchType.LAZY,           // 延遲載入出版商資訊
        optional=false                  // 書籍必須有出版商（不可為null）
    )
    @JoinColumn(
        name="publisher_id",            // 外鍵欄位名稱
        referencedColumnName="publisher_id",  // 參照的主鍵欄位
        nullable=false                  // 外鍵不可為null
    )
    @JsonBackReference                  // 解決JSON循環參照問題的被控端
    private Publisher publisher;

    // 建構子
    public Book() {
    }

    public Book(String title, String author) {
        this.title = title;
        this.author = author;
    }

    public Book(String title, String author, String isbn, BigDecimal price, 
               LocalDate publishDate, String category) {
        this.title = title;
        this.author = author;
        this.isbn = isbn;
        this.price = price;
        this.publishDate = publishDate;
        this.category = category;
    }

    // === Getter 和 Setter 方法 ===
    public int getBookId() {
        return this.bookId;
    }

    public void setBookId(int bookId) {
        this.bookId = bookId;
    }

    public String getTitle() {
        return this.title;
    }

    public void setTitle(String title) {
        this.title = title;
    }

    public String getAuthor() {
        return this.author;
    }

    public void setAuthor(String author) {
        this.author = author;
    }

    public String getIsbn() {
        return this.isbn;
    }

    public void setIsbn(String isbn) {
        this.isbn = isbn;
    }

    public BigDecimal getPrice() {
        return this.price;
    }

    public void setPrice(BigDecimal price) {
        this.price = price;
    }

    public LocalDate getPublishDate() {
        return this.publishDate;
    }

    public void setPublishDate(LocalDate publishDate) {
        this.publishDate = publishDate;
    }

    public String getCategory() {
        return this.category;
    }

    public void setCategory(String category) {
        this.category = category;
    }

    public Integer getPageCount() {
        return this.pageCount;
    }

    public void setPageCount(Integer pageCount) {
        this.pageCount = pageCount;
    }

    public String getDescription() {
        return this.description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public Integer getStockQuantity() {
        return this.stockQuantity;
    }

    public void setStockQuantity(Integer stockQuantity) {
        this.stockQuantity = stockQuantity;
    }

    public Publisher getPublisher() {
        return this.publisher;
    }

    public void setPublisher(Publisher publisher) {
        this.publisher = publisher;
    }

    // === 業務邏輯方法 ===
    
    /**
     * 取得出版商名稱（避免延遲載入問題）
     * @return 出版商名稱，如果出版商為null則返回"未知"
     */
    public String getPublisherName() {
        return this.publisher != null ? this.publisher.getPublisherName() : "未知";
    }

    /**
     * 取得出版商ID（避免延遲載入問題）
     * @return 出版商ID，如果出版商為null則返回"N/A"
     */
    public String getPublisherId() {
        return this.publisher != null ? this.publisher.getPublisherId() : "N/A";
    }

    /**
     * 檢查書籍是否有庫存
     * @return 是否有庫存
     */
    public boolean isInStock() {
        return this.stockQuantity != null && this.stockQuantity > 0;
    }

    /**
     * 檢查書籍是否為新書（出版日期在6個月內）
     * @return 是否為新書
     */
    public boolean isNewBook() {
        if (this.publishDate == null) return false;
        LocalDate sixMonthsAgo = LocalDate.now().minusMonths(6);
        return this.publishDate.isAfter(sixMonthsAgo);
    }

    /**
     * 取得格式化的價格字串
     * @return 格式化的價格（例如：$450.00）
     */
    public String getFormattedPrice() {
        return this.price != null ? "NT$" + this.price.toString() : "價格未定";
    }

    /**
     * 減少庫存數量
     * @param quantity 要減少的數量
     * @return 是否成功減少庫存
     */
    public boolean reduceStock(int quantity) {
        if (this.stockQuantity == null || this.stockQuantity < quantity) {
            return false;
        }
        this.stockQuantity -= quantity;
        return true;
    }

    /**
     * 增加庫存數量
     * @param quantity 要增加的數量
     */
    public void addStock(int quantity) {
        if (this.stockQuantity == null) {
            this.stockQuantity = 0;
        }
        this.stockQuantity += quantity;
    }

    @Override
    public String toString() {
        return "Book [bookId=" + bookId + ", title=" + title + ", author=" + author 
                + ", isbn=" + isbn + ", price=" + price + ", category=" + category 
                + ", publisherId=" + getPublisherId() + "]";
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) return true;
        if (obj == null || getClass() != obj.getClass()) return false;
        Book book = (Book) obj;
        return bookId == book.bookId || (isbn != null && isbn.equals(book.isbn));
    }

    @Override
    public int hashCode() {
        return isbn != null ? isbn.hashCode() : bookId;
    }
}
```

### PublisherDAO.java - 完整的資料存取層

```java
package model;

import javax.persistence.*;
import java.util.*;
import java.math.BigDecimal;

/**
 * 出版商資料存取物件
 * 提供完整的 CRUD 操作和複雜查詢
 */
public class PublisherDAO {
    
    private static final String PERSISTENCE_UNIT = "bookstore";
    
    /**
     * 建立資料庫連線
     * @return EntityManager 實例
     */
    public EntityManager createConnection() {
        try {
            EntityManagerFactory factory = Persistence.createEntityManagerFactory(PERSISTENCE_UNIT);
            EntityManager mgr = factory.createEntityManager();
            return mgr;
        } catch(Exception ex) {
            System.out.println("資料庫連線錯誤: " + ex.getMessage());
            ex.printStackTrace();
        }
        return null;
    }
    
    /**
     * 取得所有出版商（不包含書籍資訊）
     * @return 出版商列表
     */
    public List<Publisher> getAll() {
        EntityManager mgr = createConnection();
        try {
            TypedQuery<Publisher> query = mgr.createNamedQuery("Publisher.findAll", Publisher.class);
            return query.getResultList();
        } catch(Exception ex) {
            System.out.println("查詢所有出版商錯誤: " + ex.getMessage());
            return new ArrayList<>();
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 取得所有出版商（包含書籍資訊）
     * 使用 JOIN FETCH 避免 N+1 查詢問題
     * @return 包含書籍的出版商列表
     */
    public List<Publisher> getAllWithBooks() {
        EntityManager mgr = createConnection();
        try {
            String jpql = "SELECT DISTINCT p FROM Publisher p LEFT JOIN FETCH p.books ORDER BY p.publisherName";
            TypedQuery<Publisher> query = mgr.createQuery(jpql, Publisher.class);
            return query.getResultList();
        } catch(Exception ex) {
            System.out.println("查詢出版商和書籍錯誤: " + ex.getMessage());
            return new ArrayList<>();
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 根據ID查詢出版商
     * @param publisherId 出版商ID
     * @return 出版商實體，找不到則返回null
     */
    public Publisher findById(String publisherId) {
        EntityManager mgr = createConnection();
        try {
            return mgr.find(Publisher.class, publisherId);
        } catch(Exception ex) {
            System.out.println("查詢出版商錯誤: " + ex.getMessage());
            return null;
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 根據ID查詢出版商（包含書籍）
     * @param publisherId 出版商ID
     * @return 包含書籍的出版商實體
     */
    public Publisher findByIdWithBooks(String publisherId) {
        EntityManager mgr = createConnection();
        try {
            String jpql = "SELECT p FROM Publisher p LEFT JOIN FETCH p.books WHERE p.publisherId = :id";
            TypedQuery<Publisher> query = mgr.createQuery(jpql, Publisher.class);
            query.setParameter("id", publisherId);
            return query.getSingleResult();
        } catch(NoResultException ex) {
            return null;
        } catch(Exception ex) {
            System.out.println("查詢出版商和書籍錯誤: " + ex.getMessage());
            return null;
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 根據名稱模糊查詢出版商
     * @param name 出版商名稱（支援部分匹配）
     * @return 匹配的出版商列表
     */
    public List<Publisher> findByName(String name) {
        EntityManager mgr = createConnection();
        try {
            TypedQuery<Publisher> query = mgr.createNamedQuery("Publisher.findByName", Publisher.class);
            query.setParameter("name", "%" + name + "%");
            return query.getResultList();
        } catch(Exception ex) {
            System.out.println("根據名稱查詢出版商錯誤: " + ex.getMessage());
            return new ArrayList<>();
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 根據國家查詢出版商
     * @param country 國家名稱
     * @return 該國家的出版商列表
     */
    public List<Publisher> findByCountry(String country) {
        EntityManager mgr = createConnection();
        try {
            TypedQuery<Publisher> query = mgr.createNamedQuery("Publisher.findByCountry", Publisher.class);
            query.setParameter("country", country);
            return query.getResultList();
        } catch(Exception ex) {
            System.out.println("根據國家查詢出版商錯誤: " + ex.getMessage());
            return new ArrayList<>();
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 新增出版商
     * @param publisher 出版商實體
     * @return 是否新增成功
     */
    public boolean addPublisher(Publisher publisher) {
        EntityManager mgr = createConnection();
        try {
            mgr.getTransaction().begin();
            mgr.persist(publisher);
            mgr.getTransaction().commit();
            return true;
        } catch(Exception ex) {
            System.out.println("新增出版商錯誤: " + ex.getMessage());
            if (mgr != null && mgr.getTransaction().isActive()) {
                mgr.getTransaction().rollback();
            }
            return false;
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 新增出版商和書籍（一次交易）
     * @param publisher 包含書籍的出版商實體
     * @return 是否新增成功
     */
    public boolean addPublisherWithBooks(Publisher publisher) {
        EntityManager mgr = createConnection();
        try {
            mgr.getTransaction().begin();
            
            // 確保書籍與出版商的雙向關聯正確設定
            List<Book> books = publisher.getBooks();
            if (books != null) {
                for(Book book : books) {
                    book.setPublisher(publisher);
                }
            }
            
            // 由於設定了 CASCADE，只需要 persist 出版商
            mgr.persist(publisher);
            mgr.getTransaction().commit();
            
            System.out.println("成功新增出版商: " + publisher.getPublisherName() + 
                             "，包含 " + (books != null ? books.size() : 0) + " 本書籍");
            return true;
        } catch(Exception ex) {
            System.out.println("新增出版商和書籍錯誤: " + ex.getMessage());
            ex.printStackTrace();
            if (mgr != null && mgr.getTransaction().isActive()) {
                mgr.getTransaction().rollback();
            }
            return false;
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 更新出版商資訊
     * @param publisher 要更新的出版商實體
     * @return 是否更新成功
     */
    public boolean updatePublisher(Publisher publisher) {
        EntityManager mgr = createConnection();
        try {
            mgr.getTransaction().begin();
            
            // 檢查出版商是否存在
            Publisher existing = mgr.find(Publisher.class, publisher.getPublisherId());
            if (existing == null) {
                System.out.println("出版商不存在: " + publisher.getPublisherId());
                return false;
            }
            
            mgr.merge(publisher);
            mgr.getTransaction().commit();
            return true;
        } catch(Exception ex) {
            System.out.println("更新出版商錯誤: " + ex.getMessage());
            if (mgr != null && mgr.getTransaction().isActive()) {
                mgr.getTransaction().rollback();
            }
            return false;
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 刪除出版商（會同時刪除所有相關書籍）
     * @param publisherId 出版商ID
     * @return 是否刪除成功
     */
    public boolean deletePublisher(String publisherId) {
        EntityManager mgr = createConnection();
        try {
            mgr.getTransaction().begin();
            
            Publisher publisher = mgr.find(Publisher.class, publisherId);
            if (publisher == null) {
                System.out.println("出版商不存在: " + publisherId);
                return false;
            }
            
            int bookCount = publisher.getBooks().size();
            
            // 由於設定了 CASCADE.ALL，刪除出版商會自動刪除相關書籍
            mgr.remove(publisher);
            mgr.getTransaction().commit();
            
            System.out.println("成功刪除出版商: " + publisher.getPublisherName() + 
                             "，同時刪除了 " + bookCount + " 本書籍");
            return true;
        } catch(Exception ex) {
            System.out.println("刪除出版商錯誤: " + ex.getMessage());
            if (mgr != null && mgr.getTransaction().isActive()) {
                mgr.getTransaction().rollback();
            }
            return false;
        } finally {
            if (mgr != null) mgr.close();
        }
    }
    
    /**
     * 取得指定出版商的所有書籍
     * @param publisherId 出版商ID
     * @return 該出版商的書籍列表
     */
    public List<Book> getBooksByPublisher(String publisherId) {
        EntityManager mgr = createConnection();
        try {
            String jpql = "SELECT b FROM Book b WHERE b.publisher.publisherId = :publisherId ORDER BY b.publishDate DESC";
```
