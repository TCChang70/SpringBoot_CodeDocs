```sql
FLUSH PRIVILEGES;

USE bookstore_db;

-- 建立出版商表
CREATE TABLE publishers (
    publisher_id VARCHAR(10) NOT NULL PRIMARY KEY,
    publisher_name VARCHAR(100) NOT NULL,
    address VARCHAR(200),
    city VARCHAR(50),
    country VARCHAR(50),
    phone VARCHAR(20),
    email VARCHAR(100),
    website VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 建立書籍表

    book_id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    author VARCHAR(100) NOT NULL,
    isbn VARCHAR(20) UNIQUE,
    price DECIMAL(10,2),
    publish_date DATE,
    category VARCHAR(50),
    page_count INT,
    description TEXT,
    stock_quantity INT DEFAULT 0,
    publisher_id VARCHAR(10) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (publisher_id) REFERENCES publishers(publisher_id) ON DELETE CASCADE
);

CREATE INDEX idx_books_isbn ON books(isbn);
```

```java
        publisher.setCountry("台灣");
```

### PublisherDAOTest.java

```java
package test;

import org.junit.*;
import model.*;
import javax.persistence.*;
import java.util.List;
import static org.junit.Assert.*;

public class PublisherDAOTest {
    
    private static EntityManagerFactory emf;
    private EntityManager em;
    private PublisherDAO publisherDAO;
    
    @BeforeClass
    public static void setUpClass() {
        emf = Persistence.createEntityManagerFactory("bookstore-test");
    }
    
    @AfterClass
    public static void tearDownClass() {
        if (emf != null) {
            emf.close();
        }
    }
    
    @Before
    public void setUp() {
        em = emf.createEntityManager();
        publisherDAO = new PublisherDAO();
        
        // 插入測試資料
        em.getTransaction().begin();
        
        Publisher publisher = new Publisher("TEST001", "測試出版社");
        publisher.setCity("台北");
        publisher.setCountry("台灣");
        em.persist(publisher);
        
        em.getTransaction().commit();
    }
    
    @After
    public void tearDown() {
        if (em != null) {
            em.close();
        }
    }
    
    @Test
    public void testFindById() {
        Publisher publisher = publisherDAO.findById("TEST001");
        assertNotNull(publisher);
        assertEquals("測試出版社", publisher.getPublisherName());
    }
    
    @Test
    public void testFindByIdNotFound() {
        Publisher publisher = publisherDAO.findById("NOTFOUND");
        assertNull(publisher);
    }
    
    @Test
    public void testAddPublisher() {
        Publisher newPublisher = new Publisher("TEST002", "新測試出版社");
        newPublisher.setCity("高雄");
        newPublisher.setCountry("台灣");
        
        boolean result = publisherDAO.addPublisher(newPublisher);
        assertTrue(result);
        
        Publisher found = publisherDAO.findById("TEST002");
        assertNotNull(found);
        assertEquals("新測試出版社", found.getPublisherName());
    }
    
    @Test
    public void testGetAll() {
        List<Publisher> publishers = publisherDAO.getAll();
        assertNotNull(publishers);
        assertTrue(publishers.size() >= 1);
    }
    
    @Test
    public void testFindByName() {
        List<Publisher> publishers = publisherDAO.findByName("測試");
        assertNotNull(publishers);
        assertTrue(publishers.size() >= 1);
        
        Publisher publisher = publishers.get(0);
        assertTrue(publisher.getPublisherName().contains("測試"));
    }
}
```

## 🔧 開發環境設置步驟

### 1. 資料庫設置
```bash
# 啟動 MySQL
sudo systemctl start mysql

# 登入 MySQL
mysql -u root -p

# 執行建立資料庫腳本
source /path/to/database-setup.sql
```

### 2. 專案建置
```bash
# 編譯專案
mvn clean compile

# 執行測試
mvn test

# 打包
mvn package

# 部署到 Tomcat
cp target/publisher-book-management-1.0.0.war $TOMCAT_HOME/webapps/
```

### 3. 測試 API
```bash
# 測試連線
curl -X GET http://localhost:8080/publisher-book-management/api/publishers

# 新增出版商
curl -X POST http://localhost:8080/publisher-book-management/api/publishers \
  -H "Content-Type: application/json" \
  -d '{"publisherId":"TEST001","publisherName":"測試出版社"}'
```
